\def\cuberoot#1{%
  \setbox0=\hbox{$\m@th#1\sqrtsign{3}$}%
  \dimen0=\ht0
  \advance\dimen0-0.2\ht0
  \setbox2=\hbox{\vrule height\ht0 depth -\dimen0}%
  {\box0\lower0.4pt\box2}%
}

\[
